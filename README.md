## Config File


```toml
log = "debug"
sleeptime = 1
error_post = true

[slack_update_post]
text = "Update!!!!!"
username = "TONKOTSU bot"
icon_emoji = ":pig:"
channel = "#test"
link_names = true

[slack_error_post]
text = "Error!!!!!"
username = "bot"
icon_emoji = ":ghost:"
channel = "#test"
link_names = true

[slack_start_post]

username = "bot"
icon_emoji = ":ghost:"
channel = "#test"
link_names = true

[android]
package = "com.mercariapp.mercari" # your Android application package name
```
